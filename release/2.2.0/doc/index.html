How to Use

1. Load component.js

<script src="../dist/component.js"></script>

2. Use eg.Component

class Some extends eg.Component {
  hi() {
    alert("hi");
  }
  thing() {
    this.once("hi", this.hi);
  }
}

var some = new Some();
some.thing();
some.trigger("hi");
// fire alert("hi");
some.trigger("hi");
// Nothing happens

3. Typescript Support

You can use Typescript Generics to define events that component can emit.

import Component from "@egjs/component";

type ComponentEvent = {
  event1: {
    prop1: number;
    prop2: string;
  };
  // You can also define like this if there're more than one arguments for a event
  // First argument should be object type
  event2: (arg0: {
    prop1: number;
    prop2: string;
  }, arg1: string, arg2: boolean) => void;
  // If there're no event props
  event3: void;
}

// tslint:disable-next-line max-classes-per-file
class SomeClass extends Component<ComponentEvent> {
  thing() {
    this.on("event1", e => {
      // These properties will be available for typescript-enabled environment
      e.prop1; // number
      e.prop2; // string
      e.currentTarget; // SomeClass
      e.eventType; // string("event1")
      e.stop; // () => void
    });
  }
}
